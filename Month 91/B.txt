using namespace std;
using ll = long long;

int lcm(int x, int y) {
	int res = x * y;
	while (y > 0) {
		int tmp = x;
		x = y;
		y = tmp % y;
	}
	return res / x;
}

int main() {
	std::ios::sync_with_stdio(false);
	cin.tie(nullptr);
	cout.tie(nullptr);
	
	ll t, n;
	cin >> t;
	while (t--) {
		cin >> n;
		if (n % 4 == 0) {
			cout << "Bong" << endl;
		}
		else {
			cout << "Bing" << endl;
		}
	}

	return 0;
}
